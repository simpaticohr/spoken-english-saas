import React, { useState, useEffect } from 'react';
import { useParams, useNavigate } from 'react-router-dom';
import { motion } from 'framer-motion';
import { useSpeechSynthesis } from '../../hooks/useSpeechSynthesis';
import { useSpeechRecognition } from '../../hooks/useSpeechRecognition';
import {
  ArrowLeft,
  Play,
  Pause,
  Mic,
  MicOff,
  Volume2,
  CheckCircle,
  ChevronRight,
  ChevronLeft,
  RotateCcw,
} from 'lucide-react';

const LessonDetail = () => {
  const { id } = useParams();
  const navigate = useNavigate();
  const [currentStep, setCurrentStep] = useState(0);
  const [completed, setCompleted] = useState(false);
  const { speak, stop, isSpeaking } = useSpeechSynthesis();
  const { isListening, transcript, startListening, stopListening, resetTranscript } =
    useSpeechRecognition();

  // Demo lesson content
  const lesson = {
    id,
    title: 'Greetings & Introductions',
    category: 'conversation',
    level: 'beginner',
    steps: [
      {
        type: 'listen',
        title: 'Listen & Repeat',
        instruction: 'Listen to the phrase, then repeat it.',
        phrase: 'Hello! My name is John. Nice to meet you.',
        tips: ['Speak clearly and at a natural pace', 'Focus on the intonation pattern'],
      },
      {
        type: 'speak',
        title: 'Your Turn',
        instruction: 'Now introduce yourself using a similar pattern.',
        template: 'Hello! My name is [your name]. Nice to meet you.',
        example: 'Hello! My name is Sarah. Nice to meet you.',
      },
      {
        type: 'listen',
        title: 'Common Responses',
        instruction: 'Listen to these common greeting responses.',
        phrase: "Nice to meet you too! How are you doing today?",
        tips: ['Notice the rising intonation for questions'],
      },
      {
        type: 'speak',
        title: 'Practice Response',
        instruction: 'Respond to the greeting naturally.',
        template: "I'm doing great, thank you! How about you?",
      },
      {
        type: 'conversation',
        title: 'Full Conversation',
        instruction: 'Practice the complete conversation.',
        dialogue: [
          { speaker: 'A', text: 'Hello! My name is Alex.' },
          { speaker: 'B', text: 'Hi Alex! Nice to meet you. I\'m Maria.' },
          { speaker: 'A', text: 'Nice to meet you too, Maria! How are you today?' },
          { speaker: 'B', text: "I'm doing well, thanks! And you?" },
          { speaker: 'A', text: "I'm great, thank you for asking!" },
        ],
      },
    ],
  };

  const step = lesson.steps[currentStep];

  const handleNext = () => {
    if (currentStep < lesson.steps.length - 1) {
      setCurrentStep(currentStep + 1);
      resetTranscript();
    } else {
      setCompleted(true);
    }
  };

  const handlePrev = () => {
    if (currentStep > 0) {
      setCurrentStep(currentStep - 1);
      resetTranscript();
    }
  };

  return (
    <div className="max-w-4xl mx-auto">
      {/* Header */}
      <div className="flex items-center gap-4 mb-6">
        <button onClick={() => navigate('/lessons')} className="p-2 rounded-xl hover:bg-gray-100">
          <ArrowLeft className="w-5 h-5" />
        </button>
        <div className="flex-1">
          <h1 className="text-xl font-bold text-gray-900">{lesson.title}</h1>
          <p className="text-sm text-gray-500">
            Step {currentStep + 1} of {lesson.steps.length}
          </p>
        </div>
      </div>

      {/* Progress bar */}
      <div className="h-2 bg-gray-100 rounded-full mb-8 overflow-hidden">
        <motion.div
          className="h-full bg-primary-500 rounded-full"
          animate={{ width: `${((currentStep + 1) / lesson.steps.length) * 100}%` }}
          transition={{ duration: 0.3 }}
        />
      </div>

      {completed ? (
        <motion.div
          initial={{ opacity: 0, scale: 0.95 }}
          animate={{ opacity: 1, scale: 1 }}
          className="text-center py-16"
        >
          <div className="w-20 h-20 bg-green-100 rounded-full flex items-center justify-center mx-auto mb-6">
            <CheckCircle className="w-10 h-10 text-green-600" />
          </div>
          <h2 className="text-2xl font-bold text-gray-900 mb-2">Lesson Complete! ğŸ‰</h2>
          <p className="text-gray-500 mb-8">Great job! You've finished this lesson.</p>
          <div className="flex gap-3 justify-center">
            <button onClick={() => { setCurrentStep(0); setCompleted(false); }} className="btn-secondary">
              <RotateCcw className="w-4 h-4 mr-2" />
              Practice Again
            </button>
            <button onClick={() => navigate('/lessons')} className="btn-primary">
              Next Lesson
              <ChevronRight className="w-4 h-4 ml-2" />
            </button>
          </div>
        </motion.div>
      ) : (
        <motion.div
          key={currentStep}
          initial={{ opacity: 0, x: 20 }}
          animate={{ opacity: 1, x: 0 }}
          className="card"
        >
          {/* Step header */}
          <div className="mb-6">
            <span className="text-xs font-semibold text-primary-600 uppercase tracking-wide">
              {step.type === 'listen' ? 'ğŸ§ Listen' : step.type === 'speak' ? 'ğŸ—£ï¸ Speak' : 'ğŸ’¬ Conversation'}
            </span>
            <h2 className="text-xl font-bold text-gray-900 mt-1">{step.title}</h2>
            <p className="text-gray-500 mt-1">{step.instruction}</p>
          </div>

          {/* Content based on step type */}
          {step.type === 'listen' && (
            <div className="space-y-4">
              <div className="bg-blue-50 rounded-xl p-6">
                <p className="text-lg font-medium text-blue-900">{step.phrase}</p>
              </div>
              <button
                onClick={() => (isSpeaking ? stop() : speak(step.phrase))}
                className={`flex items-center gap-2 px-5 py-2.5 rounded-xl font-medium transition-all ${
                  isSpeaking
                    ? 'bg-red-100 text-red-700 hover:bg-red-200'
                    : 'bg-primary-100 text-primary-700 hover:bg-primary-200'
                }`}
              >
                {isSpeaking ? <Pause className="w-4 h-4" /> : <Play className="w-4 h-4" />}
                {isSpeaking ? 'Stop' : 'Listen'}
              </button>
              {step.tips && (
                <div className="bg-yellow-50 rounded-xl p-4">
                  <p className="text-sm font-medium text-yellow-800 mb-2">ğŸ’¡ Tips:</p>
                  <ul className="text-sm text-yellow-700 space-y-1">
                    {step.tips.map((tip, i) => (
                      <li key={i}>â€¢ {tip}</li>
                    ))}
                  </ul>
                </div>
              )}
            </div>
          )}

          {step.type === 'speak' && (
            <div className="space-y-4">
              {step.template && (
                <div className="bg-gray-50 rounded-xl p-6">
                  <p className="text-sm text-gray-500 mb-1">Say something like:</p>
                  <p className="text-lg font-medium text-gray-800">{step.template}</p>
                </div>
              )}

              {/* Recording area */}
              <div className="bg-white border-2 border-dashed border-gray-200 rounded-xl p-8 text-center">
                <button
                  onClick={() => (isListening ? stopListening() : startListening())}
                  className={`w-16 h-16 rounded-full flex items-center justify-center mx-auto mb-4 transition-all ${
                    isListening
                      ? 'bg-red-500 text-white pulse-recording'
                      : 'bg-primary-600 text-white hover:bg-primary-700'
                  }`}
                >
                  {isListening ? <MicOff className="w-7 h-7" /> : <Mic className="w-7 h-7" />}
                </button>
                <p className="text-sm text-gray-500">
                  {isListening ? 'Listening... Tap to stop' : 'Tap to start speaking'}
                </p>

                {transcript && (
                  <div className="mt-4 p-4 bg-green-50 rounded-xl">
                    <p className="text-sm text-gray-500 mb-1">You said:</p>
                    <p className="text-lg font-medium text-green-800">{transcript}</p>
                  </div>
                )}
              </div>
            </div>
          )}

          {step.type === 'conversation' && (
            <div className="space-y-3">
              {step.dialogue.map((line, i) => (
                <div
                  key={i}
                  className={`flex ${line.speaker === 'A' ? 'justify-start' : 'justify-end'}`}
                >
                  <div
                    className={`max-w-xs sm:max-w-md p-4 rounded-2xl ${
                      line.speaker === 'A'
                        ? 'bg-gray-100 rounded-tl-none'
                        : 'bg-primary-100 rounded-tr-none'
                    }`}
                  >
                    <p className="text-xs font-medium text-gray-500 mb-1">
                      Speaker {line.speaker}
                    </p>
                    <p className="text-gray-800">{line.text}</p>
                    <button
                      onClick={() => speak(line.text)}
                      className="mt-2 text-xs text-primary-600 hover:text-primary-700 flex items-center gap-1"
                    >
                      <Volume2 className="w-3 h-3" /> Listen
                    </button>
                  </div>
                </div>
              ))}
            </div>
          )}

          {/* Navigation */}
          <div className="flex items-center justify-between mt-8 pt-6 border-t border-gray-100">
            <button
              onClick={handlePrev}
              disabled={currentStep === 0}
              className="flex items-center gap-2 px-4 py-2 rounded-xl text-gray-600 hover:bg-gray-100 disabled:opacity-30 disabled:cursor-not-allowed"
            >
              <ChevronLeft className="w-4 h-4" />
              Previous
            </button>
            <button onClick={handleNext} className="btn-primary flex items-center gap-2">
              {currentStep === lesson.steps.length - 1 ? 'Complete' : 'Next'}
              <ChevronRight className="w-4 h-4" />
            </button>
          </div>
        </motion.div>
      )}
    </div>
  );
};

export default LessonDetail;
