import React, { useState, useRef, useEffect } from 'react';
import { useNavigate } from 'react-router-dom';
import { motion, AnimatePresence } from 'framer-motion';
import { useSpeechRecognition } from '../../hooks/useSpeechRecognition';
import { useSpeechSynthesis } from '../../hooks/useSpeechSynthesis';
import { usePracticeStore } from '../../store/practiceStore';
import { SPEECH_TOPICS } from '../../utils/constants';
import {
  ArrowLeft,
  Mic,
  MicOff,
  Send,
  Volume2,
  Bot,
  User,
  RefreshCw,
  Lightbulb,
} from 'lucide-react';

const ConversationPractice = () => {
  const navigate = useNavigate();
  const [level, setLevel] = useState('intermediate');
  const [topic, setTopic] = useState('');
  const [started, setStarted] = useState(false);
  const [inputMode, setInputMode] = useState('voice'); // 'voice' or 'text'
  const [textInput, setTextInput] = useState('');
  const messagesEndRef = useRef(null);
  const { speak, isSpeaking } = useSpeechSynthesis();
  const {
    isListening,
    transcript,
    startListening,
    stopListening,
    resetTranscript,
  } = useSpeechRecognition();

  const {
    conversationMessages,
    sendConversationMessage,
    startSession,
    loading,
  } = usePracticeStore();

  const scrollToBottom = () => {
    messagesEndRef.current?.scrollIntoView({ behavior: 'smooth' });
  };

  useEffect(scrollToBottom, [conversationMessages]);

  const handleStart = () => {
    startSession('conversation');
    setStarted(true);
    // Send initial topic
    const initialMessage = topic
      ? `Let's practice a conversation about: ${topic}`
      : "Let's have a casual English conversation. Please start.";
    handleSend(initialMessage);
  };

  const handleSend = async (message) => {
    if (!message.trim()) return;
    try {
      const response = await sendConversationMessage(message.trim());
      // Auto-speak AI response
      if (response?.message) {
        speak(response.message, { rate: 0.85 });
      }
    } catch (err) {
      // Demo fallback
      const demoResponses = [
        { message: "That's great! Can you tell me more about that?", feedback: "Good sentence structure!" },
        { message: "Interesting! How do you feel about that?", feedback: "Try using more descriptive words." },
        { message: "I see. What else can you share?", feedback: "Good use of natural speech patterns." },
      ];
      const demo = demoResponses[Math.floor(Math.random() * demoResponses.length)];
      usePracticeStore.setState((state) => ({
        conversationMessages: [
          ...state.conversationMessages,
          { role: 'assistant', content: demo.message, feedback: demo.feedback },
        ],
        loading: false,
      }));
      speak(demo.message, { rate: 0.85 });
    }
    resetTranscript();
    setTextInput('');
  };

  const handleVoiceSubmit = () => {
    if (transcript.trim()) {
      handleSend(transcript);
    }
  };

  useEffect(() => {
    if (transcript && !isListening) {
      handleVoiceSubmit();
    }
  }, [isListening]);

  if (!started) {
    return (
      <div className="max-w-2xl mx-auto">
        <div className="flex items-center gap-4 mb-6">
          <button onClick={() => navigate('/practice')} className="p-2 rounded-xl hover:bg-gray-100">
            <ArrowLeft className="w-5 h-5" />
          </button>
          <div>
            <h1 className="text-2xl font-bold text-gray-900">AI Conversation</h1>
            <p className="text-gray-500">Practice speaking with an AI partner</p>
          </div>
        </div>

        <div className="card">
          <div className="text-center mb-6">
            <div className="w-16 h-16 bg-green-100 rounded-2xl flex items-center justify-center mx-auto mb-4">
              <Bot className="w-8 h-8 text-green-600" />
            </div>
            <h2 className="text-xl font-bold text-gray-900 mb-2">Choose a Topic</h2>
            <p className="text-gray-500">
              Select a conversation topic or start a free conversation
            </p>
          </div>

          {/* Level selector */}
          <div className="flex gap-2 mb-4 justify-center">
            {['beginner', 'intermediate', 'advanced'].map((l) => (
              <button
                key={l}
                onClick={() => setLevel(l)}
                className={`px-4 py-2 rounded-xl text-sm font-medium capitalize transition-colors ${
                  level === l
                    ? 'bg-primary-600 text-white'
                    : 'bg-gray-100 text-gray-600 hover:bg-gray-200'
                }`}
              >
                {l}
              </button>
            ))}
          </div>

          {/* Topics */}
          <div className="grid grid-cols-1 sm:grid-cols-2 gap-2 mb-6">
            {SPEECH_TOPICS[level]?.map((t) => (
              <button
                key={t}
                onClick={() => setTopic(t)}
                className={`text-left p-3 rounded-xl text-sm transition-colors ${
                  topic === t
                    ? 'bg-primary-100 text-primary-700 border-2 border-primary-300'
                    : 'bg-gray-50 text-gray-700 hover:bg-gray-100 border-2 border-transparent'
                }`}
              >
                {t}
              </button>
            ))}
          </div>

          <button onClick={handleStart} className="btn-primary w-full">
            Start Conversation
          </button>
        </div>
      </div>
    );
  }

  return (
    <div className="max-w-2xl mx-auto flex flex-col" style={{ height: 'calc(100vh - 160px)' }}>
      {/* Header */}
      <div className="flex items-center gap-4 mb-4">
        <button onClick={() => { setStarted(false); }} className="p-2 rounded-xl hover:bg-gray-100">
          <ArrowLeft className="w-5 h-5" />
        </button>
        <div className="flex items-center gap-3">
          <div className="w-10 h-10 bg-green-100 rounded-xl flex items-center justify-center">
            <Bot className="w-5 h-5 text-green-600" />
          </div>
          <div>
            <h2 className="font-semibold text-gray-900">AI Conversation Partner</h2>
            <p className="text-xs text-gray-400">{topic || 'Free conversation'}</p>
          </div>
        </div>
      </div>

      {/* Messages */}
      <div className="flex-1 overflow-y-auto space-y-4 mb-4 p-2">
        <AnimatePresence>
          {conversationMessages.map((msg, i) => (
            <motion.div
              key={i}
              initial={{ opacity: 0, y: 10 }}
              animate={{ opacity: 1, y: 0 }}
              className={`flex ${msg.role === 'user' ? 'justify-end' : 'justify-start'}`}
            >
              <div className={`max-w-[80%] ${msg.role === 'user' ? 'order-1' : ''}`}>
                <div
                  className={`p-4 rounded-2xl ${
                    msg.role === 'user'
                      ? 'bg-primary-600 text-white rounded-tr-none'
                      : 'bg-white border border-gray-100 text-gray-800 rounded-tl-none shadow-sm'
                  }`}
                >
                  <p>{msg.content}</p>
                </div>
                {/* Feedback from AI */}
                {msg.feedback && (
                  <div className="mt-1 px-3 py-1.5 bg-yellow-50 rounded-lg flex items-start gap-1.5">
                    <Lightbulb className="w-3.5 h-3.5 text-yellow-600 mt-0.5 shrink-0" />
                    <p className="text-xs text-yellow-700">{msg.feedback}</p>
                  </div>
                )}
                {/* Listen button for AI messages */}
                {msg.role === 'assistant' && (
                  <button
                    onClick={() => speak(msg.content, { rate: 0.85 })}
                    className="mt-1 text-xs text-gray-400 hover:text-primary-600 flex items-center gap-1 px-3"
                  >
                    <Volume2 className="w-3 h-3" />
                    Listen
                  </button>
                )}
              </div>
            </motion.div>
          ))}
        </AnimatePresence>

        {loading && (
          <div className="flex justify-start">
            <div className="bg-white border border-gray-100 rounded-2xl rounded-tl-none p-4 shadow-sm">
              <div className="flex gap-1">
                <div className="w-2 h-2 bg-gray-400 rounded-full animate-bounce" />
                <div className="w-2 h-2 bg-gray-400 rounded-full animate-bounce" style={{ animationDelay: '0.1s' }} />
                <div className="w-2 h-2 bg-gray-400 rounded-full animate-bounce" style={{ animationDelay: '0.2s' }} />
              </div>
            </div>
          </div>
        )}
        <div ref={messagesEndRef} />
      </div>

      {/* Input area */}
      <div className="bg-white border border-gray-200 rounded-2xl p-3">
        {/* Mode toggle */}
        <div className="flex gap-2 mb-3">
          <button
            onClick={() => setInputMode('voice')}
            className={`flex items-center gap-1.5 px-3 py-1.5 rounded-lg text-xs font-medium transition-colors ${
              inputMode === 'voice' ? 'bg-primary-100 text-primary-700' : 'text-gray-500 hover:bg-gray-100'
            }`}
          >
            <Mic className="w-3.5 h-3.5" /> Voice
          </button>
          <button
            onClick={() => setInputMode('text')}
            className={`flex items-center gap-1.5 px-3 py-1.5 rounded-lg text-xs font-medium transition-colors ${
              inputMode === 'text' ? 'bg-primary-100 text-primary-700' : 'text-gray-500 hover:bg-gray-100'
            }`}
          >
            <Send className="w-3.5 h-3.5" /> Text
          </button>
        </div>

        {inputMode === 'voice' ? (
          <div className="flex items-center gap-3">
            <button
              onClick={() => (isListening ? stopListening() : startListening())}
              className={`w-12 h-12 rounded-xl flex items-center justify-center transition-all ${
                isListening
                  ? 'bg-red-500 text-white pulse-recording'
                  : 'bg-primary-600 text-white hover:bg-primary-700'
              }`}
            >
              {isListening ? <MicOff className="w-5 h-5" /> : <Mic className="w-5 h-5" />}
            </button>
            <div className="flex-1">
              {isListening ? (
                <p className="text-sm text-red-600 font-medium">Listening...</p>
              ) : transcript ? (
                <p className="text-sm text-gray-700">{transcript}</p>
              ) : (
                <p className="text-sm text-gray-400">Tap the mic to speak</p>
              )}
            </div>
          </div>
        ) : (
          <form
            onSubmit={(e) => {
              e.preventDefault();
              handleSend(textInput);
            }}
            className="flex gap-2"
          >
            <input
              value={textInput}
              onChange={(e) => setTextInput(e.target.value)}
              placeholder="Type your message..."
              className="input-field flex-1"
            />
            <button type="submit" disabled={!textInput.trim()} className="btn-primary px-4">
              <Send className="w-4 h-4" />
            </button>
          </form>
        )}
      </div>
    </div>
  );
};

export default ConversationPractice;
